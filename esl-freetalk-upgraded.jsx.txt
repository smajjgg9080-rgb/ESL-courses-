// ─── AI CONVERSATION ───────────────────────────────────────────────────────────
async function callClaude(messages, systemPrompt) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 1000,
      system: systemPrompt,
      messages,
    }),
  });
  const data = await res.json();
  return data.content?.[0]?.text || "Sorry, I couldn't respond right now.";
}

function Timer() {
  const [secs, setSecs] = useState(10 * 60);
  const [running, setRunning] = useState(false);
  const ref = useRef(null);
  useEffect(() => {
    if (running) { ref.current = setInterval(() => setSecs(s => Math.max(0, s - 1)), 1000); }
    else clearInterval(ref.current);
    return () => clearInterval(ref.current);
  }, [running]);
  const m = String(Math.floor(secs / 60)).padStart(2, "0");
  const s = String(secs % 60).padStart(2, "0");
  const pct = (secs / 600) * 100;
  return (
    <div style={{ background: "#0f172a", borderRadius: 16, padding: "1.25rem", marginBottom: "1.5rem", textAlign: "center" }}>
      <div style={{
        fontSize: "2.6rem", fontWeight: 800, color: secs < 60 ? "#ef4444" : "#38bdf8",
        fontVariantNumeric: "tabular-nums", letterSpacing: 3,
        transition: "color 0.3s",
      }}>{m}:{s}</div>
      <div style={{ background: "#1e293b", borderRadius: 50, height: 6, margin: "0.75rem 0", overflow: "hidden" }}>
        <div style={{
          height: "100%", borderRadius: 50, transition: "width 1s linear",
          width: `${pct}%`,
          background: secs < 60 ? "#ef4444" : "linear-gradient(90deg,#38bdf8,#818cf8)",
        }} />
      </div>
      <div style={{ display: "flex", justifyContent: "center", gap: 8, marginTop: "0.6rem" }}>
        {[
          { label: "▶ Start", fn: () => setRunning(true), bg: "#10b981" },
          { label: "⏸ Pause", fn: () => setRunning(false), bg: "#f59e0b" },
          { label: "↺ Reset", fn: () => { setRunning(false); setSecs(600); }, bg: "#ef4444" },
          { label: "5 min", fn: () => { setRunning(false); setSecs(300); }, bg: "#6366f1" },

function AIChat({ topic }) {
  const [msgs, setMsgs] = useState([]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef(null);
  const system = `You are an ESL conversation partner helping a student practice English.
The topic is "${topic.title}". Keep responses encouraging, 2–3 sentences max.
Correct any grammar errors gently. Ask a follow-up question. Be warm and friendly.`;

  const send = async () => {
    if (!input.trim() || loading) return;
    const newMsgs = [...msgs, { role: "user", content: input }];
    setMsgs(newMsgs); setInput(""); setLoading(true);
    const reply = await callClaude(newMsgs, system);
    setMsgs([...newMsgs, { role: "assistant", content: reply }]);
    setLoading(false);
  };

  useEffect(() => { bottomRef.current?.scrollIntoView({ behavior: "smooth" }); }, [msgs, loading]);

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
      <div style={{ background: "#f8fafc", borderRadius: 16, padding: "1rem", minHeight: 220, maxHeight: 300, overflowY: "auto" }}>
        {msgs.length === 0 && (
          <div style={{ color: "#94a3b8", fontSize: "0.85rem", textAlign: "center", paddingTop: 60 }}>
            👋 Start speaking about <strong>{topic.title}</strong>!<br />Type anything to begin.
          </div>
        )}
        {msgs.map((m, i) => (
          <div key={i} style={{ display: "flex", justifyContent: m.role === "user" ? "flex-end" : "flex-start", marginBottom: 8 }}>
            {m.role === "assistant" && (
              <span style={{ marginRight: 6, fontSize: "1.3rem" }}>🤖</span>
            )}
            <div style={{
              padding: "0.65rem 1rem", borderRadius: 14, maxWidth: "78%", fontSize: "0.88rem", lineHeight: 1.55,
              background: m.role === "user" ? GRADIENTS[topic.gradient] : "white",
              color: m.role === "user" ? "white" : "#1e293b",
              boxShadow: "0 2px 8px rgba(0,0,0,0.08)",
            }}>{m.content}</div>
          </div>
        ))}
        {loading && (
          <div style={{ display: "flex", gap: 6, padding: "0.5rem" }}>
            {[0,1,2].map(i => (
              <div key={i} style={{
                width: 8, height: 8, borderRadius: "50%", background: "#6366f1",
                animation: `bounce 1.2s ease-in-out ${i * 0.2}s infinite`,
              }} />
            ))}
          </div>
        )}
        <div ref={bottomRef} />
      </div>
      <div style={{ display: "flex", gap: 8 }}>
        <input
          value={input} onChange={e => setInput(e.target.value)}
          onKeyDown={e => e.key === "Enter" && send()}
          placeholder="Type your answer or question…"
          style={{
            flex: 1, padding: "0.7rem 1rem", borderRadius: 12,
            border: "2px solid #e2e8f0", fontFamily: "inherit", fontSize: "0.88rem",
            outline: "none",
          }}
        />
        <button onClick={send} disabled={loading} style={{
          padding: "0.7rem 1.2rem", background: GRADIENTS[topic.gradient],
          border: "none", borderRadius: 12, color: "white",
          fontWeight: 700, cursor: "pointer", fontFamily: "inherit",
          opacity: loading ? 0.6 : 1,
        }}>Send</button>
      </div>
      <style>{`@keyframes bounce{0%,100%{transform:translateY(0)}50%{transform:translateY(-8px)}}`}</style>
    </div>
  );
}

                    background: "linear-gradient(90deg,#43e97b,#38f9d7)",
                  }} />
                </div>
              </div>
            </div>

            {/* GRID */}
            {filtered.length === 0 ? (
              <div style={{ textAlign: "center", padding: "4rem", color: "#94a3b8" }}>
                <div style={{ fontSize: "3rem", marginBottom: "1rem" }}>🔍</div>
                <p style={{ fontWeight: 700 }}>No topics found for "{search}"</p>
              </div>
            ) : (
              <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill,minmax(290px,1fr))", gap: "1.5rem" }}>
                {filtered.map(t => (
                  <div key={t.id} style={{ position: "relative" }}>
                    {completed.has(t.id) && (
                      <div style={{
                        position: "absolute", top: 12, right: 12, zIndex: 10,
                        background: "#10b981", color: "white", borderRadius: "50%",
                        width: 28, height: 28, display: "flex", alignItems: "center", justifyContent: "center",
                        fontSize: "0.9rem", fontWeight: 800, boxShadow: "0 2px 8px rgba(16,185,129,0.4)",
                      }}>✓</div>
                    )}
                    <TopicCard topic={t} onClick={handleTopicOpen} />
                  </div>
                ))}
              </div>
            )}
          </div>
        )}

        {section === "offline" && <OfflineSection />}
      </div>

      {activeTopic && <TopicModal topic={activeTopic} onClose={() => setActiveTopic(null)} />}
    </div>
  );
}
